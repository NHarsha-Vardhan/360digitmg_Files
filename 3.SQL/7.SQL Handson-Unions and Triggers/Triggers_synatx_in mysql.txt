DELIMITER $$
CREATE TRIGGER  cal_tot_insert
BEFORE INSERT ON sales_data1
FOR EACH ROW
BEGIN
    SET NEW.total_price = NEW.sales * NEW.quantity;
END $$
DELIMITER ;

create table student_info (
stud_id int not null,
stud_code varchar(20) default null,
stud_name varchar(35) default null,
subject varchar(25) default null,
marks int default null,
phone varchar(15) default null,
primary key(stud_id)
);

#creating trigger
DELIMITER $$
CREATE TRIGGER after_insert_details
AFTER INSERT ON student_info
FOR EACH ROW
BEGIN 
	INSERT INTO student_detail 
    VALUES (new.stud_id,new.stud_code,new.stud_name,new.subject,new.marks,new.phone,CURTIME());
END $$
DELIMITER ;
